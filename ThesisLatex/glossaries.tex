\newglossaryentry{ringens}{
	name = {E. ringens},
	first = {Engraulis ringens},
	description = {A fish of the Engraulidae family, distributed in the southeastern Pacific Ocean off the coasts of Peru and Chile, commonly known as Peruvian anchovy}
}

\newglossaryentry{encrasicolus}{
	name = {E. encrasicolus},
	first = {Engraulis encrasicolus},
	description = {A fish of the Engraulidae family, distributed in the Mediterranean Sea and on the Atlantic coast of Africa and Europe, commonly known as European anchovy}
}

\newglossaryentry{ich}{
	name = {Ichthyop},
	description = {A Lagrangian tool for modelling ichthyoplankton dynamics}
}

\newglossaryentry{ich-deb}{
	name = {Ichthyop-DEB},
	description = {A Lagrangian tool for modelling ichthyoplankton dynamics, coupled to a growth module, based on the DEB theory}
}

\newglossaryentry{cri1}{
	name = {criterion 1},
	description = {Larval recruitment criterion based on retention of particles within the continental shelf (the 2000 m isobath) on a given age (30 days)}
}
